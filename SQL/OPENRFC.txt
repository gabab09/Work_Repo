select
a.rfc_id as 'RFC ID',
b.ticket_id as 'Ticket ID',
a.name as 'Description',
(select x.Username from auth.LOGIN x where x.LoginID = a.initiator_id) as 'Initiator',
(select x.Username from auth.LOGIN x where x.LoginID = c.owner_id) as 'Owner',
(select x.Name from IPim.Users x, IPim.Tickets y,IPradar.ipim_ticket_mapping z where x.id = y.LastUpdatedBy and y.id = z.ipim_id and z.ticket_id = b.ticket_id  ) as 'Last Updated User',
(select SUBSTR(x.Content,1,150) from IPim.Attachments x, IPim.Transactions y,IPradar.ipim_ticket_mapping z where x.TransactionId = y.id and y.objectid = z.ipim_id and z.ticket_id = b.ticket_id order by x.id desc LIMIT 1) as 'Last Update',
TIMESTAMPDIFF(DAY, c.update_date , NOW()) as 'Updated Ago',
c.create_date as 'Created Date',
c.update_date as 'Updated Date'
from
IPcm.rfc a,IPradar.rfc_ticket_mapping b,IPradar.tickets c
where a.rfc_id = b.rfc_id and b.ticket_id = c.ticket_id and a.workflow_id in (95,89,83,119) and a.current_version_id is NULL
order by TIMESTAMPDIFF(DAY, c.update_date , NOW() ) desc
